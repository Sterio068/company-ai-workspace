```python
# backend/accounting/main.py(新增 endpoint)
import httpx, asyncio

FAL_QUEUE_URL = "https://queue.fal.run/fal-ai/recraft-v3"
FAL_KEY = os.getenv("FAL_API_KEY", "")

class RecraftRequest(BaseModel):
    project_id: Optional[str] = None
    prompt: str
    image_size: Literal["square_hd", "portrait_16_9"] = "square_hd"
    style: str = "realistic_image"
    regenerate_of: Optional[str] = None

@app.post("/design/recraft")
async def design_recraft(req: RecraftRequest, email: Optional[str] = Depends(current_user_email)):
    if not FAL_KEY:
        raise HTTPException(503, "Fal.ai 未設定 (FAL_API_KEY 缺)")
    headers = {"Authorization": f"Key {FAL_KEY}", "Content-Type": "application/json"}
    payload = {
        "prompt": req.prompt,
        "image_size": req.image_size,
        "style": req.style,
        "num_images": 3,  # 老闆 Q2 · 一次 3 張挑方向
    }
    async with httpx.AsyncClient(timeout=15) as c:
        try:
            q = await c.post(FAL_QUEUE_URL, headers=headers, json=payload)
            q.raise_for_status()
            req_id = q.json()["request_id"]
        except httpx.HTTPStatusError as e:
            if e.response.status_code in (400, 422):
                return {"status": "rejected",
                        "friendly_message": "描述太像真人、官方標誌或敏感文字,請改成抽象視覺描述"}
            raise HTTPException(502, "Fal.ai 服務忙碌中")
        # Poll up to 12 秒
        for _ in range(12):
            s = await c.get(f"{FAL_QUEUE_URL}/requests/{req_id}/status", headers=headers)
            if s.json().get("status") == "COMPLETED":
                r = await c.get(f"{FAL_QUEUE_URL}/requests/{req_id}", headers=headers)
                # 記 log
                db.design_jobs.insert_one({
                    "request_id": req_id, "user": email, "project_id": req.project_id,
                    "prompt": req.prompt, "size": req.image_size,
                    "created_at": datetime.utcnow(),
                })
                return {"job_id": req_id, "status": "done",
                        "images": r.json().get("images", []), "friendly_message": None}
            await asyncio.sleep(1)
        return {"job_id": req_id, "status": "pending",
                "friendly_message": "生圖中,請稍候,可關掉視窗稍後查看歷史"}
```

```python
# backend/accounting/main.py(新增 endpoint · 需 pip install pymupdf)
import fitz  # pymupdf
from fastapi import UploadFile, File

@app.post("/pdf/extract")
async def pdf_extract(file: UploadFile = File(...)):
    if file.content_type not in ("application/pdf",):
        raise HTTPException(400, "請上傳 PDF 檔案")
    pdf_bytes = await file.read()
    if len(pdf_bytes) > 30 * 1024 * 1024:
        raise HTTPException(413, "PDF 超過 30 MB · 請拆檔")

    started = datetime.utcnow()
    try:
        doc = fitz.open(stream=pdf_bytes, filetype="pdf")
    except Exception:
        raise HTTPException(400, "PDF 無法解析(可能加密或損毀)")

    pages = []
    ocr_count = 0
    for i, page in enumerate(doc):
        text = page.get_text("text").strip()
        ocr_used = False
        # 文字密度 threshold:< 120 字 + 有圖 = 偏掃描頁 · 走 OCR
        if len(text) < 120 and (page.get_images() or len(page.get_text("words")) < 30):
            try:
                tp = page.get_textpage_ocr(language="chi_tra+eng")
                text = page.get_text(textpage=tp).strip()
                ocr_used = True
                ocr_count += 1
            except Exception:
                pass  # OCR 失敗不擋 · 留原 text(可能空)
        pages.append({"page": i + 1, "text": text, "ocr_used": ocr_used,
                      "char_count": len(text)})

    combined = "\n\n".join([f"=== 第 {p['page']} 頁 ===\n{p['text']}" for p in pages])
    return {
        "stats": {
            "page_count": len(pages),
            "ocr_pages": ocr_count,
            "extract_seconds": (datetime.utcnow() - started).total_seconds(),
        },
        "pages": pages,
        "combined_text": combined,
    }
```

```python
def reset_cache():
    """測試用 · 每 test 前呼叫"""
    _SCHEMA_CACHED.update({"checked": False, "ok": False, "issue": ""})

def probe_tx_schema(db) -> dict:
    if _SCHEMA_CACHED["checked"]:
        return _SCHEMA_CACHED
    _SCHEMA_CACHED["checked"] = True
    try:
        doc = db.transactions.find_one({}, {"rawAmount": 1, "model": 1, "user": 1, "createdAt": 1})
        if not doc:

            raw = doc.get("rawAmount") or {}
            if not isinstance(raw, dict): missing.append("rawAmount(non-dict)")
            elif "prompt" not in raw and "completion" not in raw:
                missing.append("rawAmount.prompt|completion")
            for k in ("model", "user", "createdAt"):
                if k not in doc: missing.append(k)
            _SCHEMA_CACHED["ok"] = not missing
            _SCHEMA_CACHED["issue"] = f"缺欄位:{missing}" if missing else ""
    except Exception as e:
        _SCHEMA_CACHED["ok"] = False
        _SCHEMA_CACHED["issue"] = f"probe 失敗:{e}"
    return _SCHEMA_CACHED

    usd = (tin / 1_000_000) * p["input"] + (tout / 1_000_000) * p["output"]
```

```python
"""NAS 檔案抽字 · 依副檔名路由 · 統一回 {path, content_preview, metadata}"""
import fitz  # pymupdf
from docx import Document
from pptx import Presentation
from openpyxl import load_workbook
from PIL import Image
from PIL.ExifTags import TAGS


def extract_pdf(path: str) -> dict:
    doc = fitz.open(path)
    pages = []
    for i, page in enumerate(doc):
        text = page.get_text("text").strip()
        if len(text) < 120 and page.get_images():
            tp = page.get_textpage_ocr(language="chi_tra+eng")
            text = page.get_text(textpage=tp).strip()
        pages.append(text)
    return {
        "type": "pdf", "page_count": len(doc),
        "content_preview": "\n\n".join(pages)[:2000],
    }


def extract_docx(path: str) -> dict:
    doc = Document(path)
    text = "\n".join([p.text for p in doc.paragraphs])
    return {"type": "docx", "content_preview": text[:2000]}


def extract_pptx(path: str) -> dict:
    prs = Presentation(path)
    slides = []
    for s in prs.slides:
        for shape in s.shapes:
            if hasattr(shape, "text"):
                slides.append(shape.text)
    return {"type": "pptx", "slide_count": len(prs.slides), "content_preview": "\n\n".join(slides)[:2000]}


def extract_xlsx(path: str) -> dict:
    wb = load_workbook(path, read_only=True, data_only=True)
    preview = []
    for sheet_name in wb.sheetnames[:3]:
        ws = wb[sheet_name]
        rows = []
        for i, row in enumerate(ws.iter_rows(max_row=20, values_only=True)):
            rows.append(" | ".join(str(c) for c in row if c is not None))
            if i >= 19: break
        preview.append(f"[{sheet_name}]\n" + "\n".join(rows))
    return {"type": "xlsx", "sheet_count": len(wb.sheetnames), "content_preview": "\n\n".join(preview)[:2000]}


def extract_image(path: str) -> dict:
    """設計圖只抽 metadata · 不抽內容(v2 再用 CLIP vision)"""
    img = Image.open(path)
    exif = {}
    if hasattr(img, "_getexif") and img._getexif():
        exif = {TAGS.get(k, k): str(v)[:100] for k, v in img._getexif().items()}
    return {
        "type": "image",
        "width": img.width, "height": img.height, "format": img.format,
        "exif": exif,
        "content_preview": f"圖片 {img.width}x{img.height} · {img.format}",
    }


EXTRACTORS = {
    ".pdf": extract_pdf, ".docx": extract_docx, ".pptx": extract_pptx,
    ".xlsx": extract_xlsx, ".xls": extract_xlsx,
    ".jpg": extract_image, ".jpeg": extract_image, ".png": extract_image,
    ".tiff": extract_image, ".webp": extract_image,
    # AI/PSD 格式 Pillow 讀不了 · 只記檔名與大小
}


def extract(path: str) -> dict:
    import os, pathlib
    ext = pathlib.Path(path).suffix.lower()
    try:
        if ext in EXTRACTORS:
            result = EXTRACTORS[ext](path)
        else:
            # 不支援格式(AI/PSD/AE 等)· 只記 metadata
            result = {"type": "unknown", "content_preview": f"檔案 {pathlib.Path(path).name}"}
        stat = os.stat(path)
        result.update({
            "path": path, "size": stat.st_size,
            "modified_at": stat.st_mtime,
            "filename": pathlib.Path(path).name,
        })
        return result
    except Exception as e:
        return {"path": path, "type": "error", "content_preview": "", "error": str(e)}
```

### 5. 索引 cron(`backend/accounting/knowledge/indexer.py`)

```python
"""每日凌晨 02:00 跑 · 遍歷所有 enabled sources · 各別增量索引"""
import os, fnmatch, pathlib, hashlib
from datetime import datetime
from .extract import extract
import meilisearch

ms = meilisearch.Client("http://meilisearch:7700", os.getenv("MEILI_MASTER_KEY"))
index = ms.index("chengfu_knowledge")  # 改名 · 不只 NAS


def _match_excluded(rel_path: str, patterns: list[str]) -> bool:
    """判斷 rel_path 是否該被排除(支援 * 與 / 開頭的目錄 pattern)"""
    name = os.path.basename(rel_path)
    for p in patterns:
        if p.startswith("/"):
            # 目錄 pattern · 例如 /sensitive/* 表示「以這層為根」
            if rel_path.startswith(p.lstrip("/").rstrip("*").rstrip("/")):
                return True
        elif fnmatch.fnmatch(name, p):
            return True
    return False


def reindex_source(source_id: str) -> dict:
    """單一 source 增量索引 · 從 mongo 讀 source 設定"""
    from .. import knowledge_sources_col  # main.py 注入的 collection
    from bson import ObjectId
    src = knowledge_sources_col.find_one({"_id": ObjectId(source_id)})
    if not src or not src.get("enabled"):
        return {"skipped": True, "reason": "停用或不存在"}

    started = datetime.utcnow()
    since_mtime = (src.get("last_indexed_at") or datetime(2000, 1, 1)).timestamp()
    excludes = src.get("exclude_patterns", [])
    mime_white = src.get("mime_whitelist")
    max_bytes = (src.get("max_size_mb", 50)) * 1024 * 1024

    docs, errors = [], 0
    for root, dirs, files in os.walk(src["path"]):
        # 修剪目錄 · 別走進排除的 dir
        dirs[:] = [d for d in dirs
                   if not _match_excluded(os.path.relpath(os.path.join(root, d), src["path"]), excludes)]
        for f in files:
            path = os.path.join(root, f)
            rel = os.path.relpath(path, src["path"])
            if _match_excluded(rel, excludes): continue
            try:
                stat = os.stat(path)
                if stat.st_mtime < since_mtime: continue
                if stat.st_size > max_bytes: continue
                ext = pathlib.Path(f).suffix.lower().lstrip(".")
                if mime_white and ext not in mime_white: continue
                doc = extract(path)
                # 衍生欄位
                doc["id"] = hashlib.md5(f"{source_id}::{rel}".encode()).hexdigest()
                doc["source_id"] = str(src["_id"])
                doc["source_name"] = src["name"]
                doc["rel_path"] = rel
                # 自動推 project(若 path 結構是 projects/<案>/...)
                parts = rel.split(os.sep)
                if parts[0] == "projects" and len(parts) > 1:
                    doc["project"] = parts[1]
                docs.append(doc)
            except Exception:
                errors += 1
                continue

    if docs:
        index.add_documents(docs)

    stats = {
        "file_count": len(docs),
        "index_seconds": (datetime.utcnow() - started).total_seconds(),
        "errors": errors,
    }
    knowledge_sources_col.update_one(
        {"_id": src["_id"]},
        {"$set": {"last_indexed_at": datetime.utcnow(), "last_index_stats": stats}}
    )
    return stats


def reindex_all() -> dict:
    """cron 每日跑 · 所有 enabled sources 各跑一輪"""
    from .. import knowledge_sources_col
    results = {}
    for src in knowledge_sources_col.find({"enabled": True}):
        results[src["name"]] = reindex_source(str(src["_id"]))
    return results
```

```python
@app.get("/knowledge/search")
def knowledge_search(q: str,
                     source_id: Optional[str] = None,    # 限定某 source
                     project: Optional[str] = None,       # 限定某專案
                     limit: int = 20,
                     email: str = Depends(current_user_email)):

    filters = []
    if source_id: filters.append(f'source_id = "{source_id}"')
    if project:   filters.append(f'project = "{project}"')
    res = index.search(q, {
        "limit": limit,
        "filter": " AND ".join(filters) if filters else None,
        "attributesToRetrieve": ["filename", "rel_path", "source_id", "source_name",
                                  "project", "type", "content_preview", "modified_at"],
    })
    return res


@app.get("/knowledge/read")
def knowledge_read(source_id: str, rel_path: str,
                   email: str = Depends(current_user_email)):
    """讀某 source 內某檔的純文字 · 給 Agent 引用歷史檔
    用 source_id + rel_path 取代 path · 強制 path traversal 防護
    """
    src = knowledge_sources_col.find_one({"_id": ObjectId(source_id), "enabled": True})
    if not src:
        raise HTTPException(404, "資料源不存在或已停用")
    # 防 path traversal
    abs_path = os.path.abspath(os.path.join(src["path"], rel_path))
    if not abs_path.startswith(os.path.abspath(src["path"])):
        raise HTTPException(403, "路徑越界")
    if not os.path.exists(abs_path):
        raise HTTPException(404, "檔案不存在")
    # Agent 存取權限檢查(若呼叫者帶 X-Agent-Num header)
    agent_num = request.headers.get("X-Agent-Num")
    if agent_num and src.get("agent_access") and agent_num not in src["agent_access"]:
        raise HTTPException(403, f"此資料源未開放給 #{agent_num} 助手")
    return extract(abs_path)


@app.get("/knowledge/list")
def knowledge_list(source_id: Optional[str] = None,
                   project: Optional[str] = None,
                   email: str = Depends(current_user_email)):
    """列 sources / 列某 source 下所有 projects / 列某專案下所有檔"""
    if not source_id:
        # 列所有 enabled sources
        srcs = list(knowledge_sources_col.find({"enabled": True}, {"_id": 1, "name": 1, "type": 1}))
        return {"sources": [{"id": str(s["_id"]), "name": s["name"], "type": s["type"]} for s in srcs]}
    src = knowledge_sources_col.find_one({"_id": ObjectId(source_id), "enabled": True})
    if not src: raise HTTPException(404)
    if project:
        folder = os.path.join(src["path"], "projects", project)
        if not os.path.exists(folder): raise HTTPException(404)
        return {"files": [extract(os.path.join(folder, f)) for f in os.listdir(folder)
                          if not f.startswith(".")]}
    projects_dir = os.path.join(src["path"], "projects")
    return {"projects": sorted(os.listdir(projects_dir)) if os.path.exists(projects_dir) else []}
```
